        const AudioSys = {
            ctx: null, isPlayingBGM: false,
            init: function() {
                if (!this.ctx) this.ctx = new (window.AudioContext || window.webkitAudioContext)();
                if (this.ctx.state === 'suspended') this.ctx.resume();
            },
            playSound: function(type) {
                if (!this.ctx) return;
                const osc = this.ctx.createOscillator();
                const gain = this.ctx.createGain();
                osc.connect(gain); gain.connect(this.ctx.destination);
                const now = this.ctx.currentTime;
                
                if (type === 'jump') { osc.type = 'sine'; osc.frequency.setValueAtTime(400, now); osc.frequency.exponentialRampToValueAtTime(800, now + 0.2); gain.gain.setValueAtTime(0.3, now); gain.gain.exponentialRampToValueAtTime(0.01, now + 0.2); osc.start(now); osc.stop(now + 0.2); } 
                else if (type === 'duck') { osc.type = 'triangle'; osc.frequency.setValueAtTime(400, now); osc.frequency.exponentialRampToValueAtTime(200, now + 0.2); gain.gain.setValueAtTime(0.3, now); gain.gain.exponentialRampToValueAtTime(0.01, now + 0.2); osc.start(now); osc.stop(now + 0.2); }
                else if (type === 'coin') { osc.type = 'square'; osc.frequency.setValueAtTime(987.77, now); osc.frequency.setValueAtTime(1318.51, now + 0.05); gain.gain.setValueAtTime(0.1, now); gain.gain.linearRampToValueAtTime(0, now + 0.15); osc.start(now); osc.stop(now + 0.15); }
                else if (type === 'crash') { osc.type = 'sawtooth'; osc.frequency.setValueAtTime(150, now); osc.frequency.exponentialRampToValueAtTime(20, now + 0.3); gain.gain.setValueAtTime(0.4, now); gain.gain.exponentialRampToValueAtTime(0.01, now + 0.3); osc.start(now); osc.stop(now + 0.3); }
            },
            playBGM: function() {
                if (!this.ctx || this.isPlayingBGM) return;
                this.isPlayingBGM = true;
                const notes = [261.63, 329.63, 392.00, 523.25]; let noteIndex = 0;
                const playNextNote = () => {
                    if (!this.isPlayingBGM) return;
                    const osc = this.ctx.createOscillator(); const gain = this.ctx.createGain();
                    osc.type = 'sine'; osc.frequency.value = notes[noteIndex];
                    osc.connect(gain); gain.connect(this.ctx.destination);
                    const now = this.ctx.currentTime;
                    gain.gain.setValueAtTime(0.05, now); gain.gain.exponentialRampToValueAtTime(0.001, now + 0.2);
                    osc.start(now); osc.stop(now + 0.2);
                    noteIndex = (noteIndex + 1) % notes.length;
                    this.bgmTimeout = setTimeout(playNextNote, 200);
                };
                playNextNote();
            },
            stopBGM: function() { this.isPlayingBGM = false; if(this.bgmTimeout) clearTimeout(this.bgmTimeout); }
        };

        function updateDayNightCycle() {
            const cycleDist = 500;
            const currentDist = distance % cycleDist;

            let targetSky, targetFog;

            if (currentDist < 100) { // เช้า
                targetSky = new THREE.Color(DAY_COLORS.sky);
                targetFog = new THREE.Color(DAY_COLORS.fog);
            } else if (currentDist < 250) { // กลางวัน
                targetSky = new THREE.Color(DAY_COLORS.sky);
                targetFog = new THREE.Color(DAY_COLORS.fog);
            } else if (currentDist < 350) { // พลบค่ำ (ค่อยๆ เปลี่ยนไปกลางคืน)
                let t = (currentDist - 250) / 100;
                targetSky = new THREE.Color(DAY_COLORS.sky).lerp(new THREE.Color(NIGHT_COLORS.sky), t);
                targetFog = new THREE.Color(DAY_COLORS.fog).lerp(new THREE.Color(NIGHT_COLORS.fog), t);
            } else { // กลางคืน
                targetSky = new THREE.Color(NIGHT_COLORS.sky);
                targetFog = new THREE.Color(NIGHT_COLORS.fog);
            }

            scene.background.lerp(targetSky, 0.05);
            scene.fog.color.lerp(targetFog, 0.05);
        }

        function isSpaceOccupied(lane, zPos, safeDistance) {
            for (let obj of gameObjects) {
                if (obj.lane === lane && Math.abs(obj.mesh.position.z - zPos) < safeDistance) {
                    return true;
                }
            }
            return false;
        }

        function spawnObstacle() {
            const types = ['rock', 'pit', 'vine'];
            const type = types[Math.floor(Math.random() * types.length)];
            const lane = Math.floor(Math.random() * 3) - 1;
            const xPos = lane * -LANE_WIDTH; 
            const zPos = 120; 
            
            // เช็คทับซ้อน ปลอดภัยไว้ที่ระยะ 8
            if (isSpaceOccupied(lane, zPos, 8)) return;

            let mesh;
            if (type === 'rock') {
                const geo = new THREE.BoxGeometry(1.6, 2.5, 1.5);
                const mat = new THREE.MeshLambertMaterial({ color: 0x808080 });
                mesh = new THREE.Mesh(geo, mat); mesh.position.set(xPos, 1.25, zPos);
                mesh.castShadow = true;
            } 
            else if (type === 'pit') {
                const geo = new THREE.BoxGeometry(1.8, 0.1, 3);
                const mat = new THREE.MeshLambertMaterial({ color: 0x1a1a1a });
                mesh = new THREE.Mesh(geo, mat); mesh.position.set(xPos, 0.05, zPos);
            }
            else if (type === 'vine') {
                // เถาวัลย์แบบใหม่: เป็นกลุ่มกระบอกห้อยลงมา
                mesh = new THREE.Group();
                const vineHeight = 2.4;
                // ปรับแกน Y ให้ความสูงฐานของเถาวัลย์อยู่ระดับ 0.8 (หนูสูง 0.6 จะลอดได้พอดี)
                const vineY = 0.8 + (vineHeight / 2); 
                
                // สร้างเส้นเถาวัลย์ 6 เส้น
                for(let j = 0; j < 6; j++) {
                    const cGeo = new THREE.CylinderGeometry(0.05, 0.05, vineHeight, 6);
                    const cMat = new THREE.MeshLambertMaterial({ color: 0x2E8B57 });
                    const c = new THREE.Mesh(cGeo, cMat);
                    c.position.set(-0.8 + (j * 0.32), vineY, 0);
                    c.rotation.z = (Math.random() - 0.5) * 0.2; // เอียงแบบสุ่มๆ
                    
                    // เพิ่มใบไม้ประดับให้เถาวัลย์
                    const leaf = new THREE.Mesh(new THREE.ConeGeometry(0.15, 0.3, 4), cMat);
                    leaf.position.set(0, (Math.random() - 0.5) * vineHeight, 0.1);
                    leaf.rotation.x = Math.PI / 2;
                    c.add(leaf);

                    c.castShadow = true;
                    mesh.add(c);
                }

                // สร้าง Hitbox แบบล่องหนเพื่อให้เช็คชนได้แม่นยำและง่ายขึ้น
                const hitGeo = new THREE.BoxGeometry(1.8, vineHeight, 0.5);
                const hitMat = new THREE.MeshBasicMaterial({visible: false});
                const hitbox = new THREE.Mesh(hitGeo, hitMat);
                hitbox.position.set(0, vineY, 0);
                mesh.add(hitbox);

                mesh.position.set(xPos, 0, zPos);
            }

            scene.add(mesh);
            gameObjects.push({ mesh: mesh, type: type, isObstacle: true, lane: lane });
        }

        function spawnCoins() {
            const lane = Math.floor(Math.random() * 3) - 1;
            const xPos = lane * -LANE_WIDTH;
            const numCoins = 5 + Math.floor(Math.random() * 3);
            const startZ = 120;
            const spacing = 2;

            for (let i = 0; i < numCoins; i++) {
                const coinZ = startZ + (i * spacing);
                // ป้องกันเหรียญทับกับอุปสรรค
                if (isSpaceOccupied(lane, coinZ, 4)) continue;

                const geo = new THREE.CylinderGeometry(0.4, 0.4, 0.1, 16);
                const mat = new THREE.MeshLambertMaterial({ color: 0xFFD700 });
                const mesh = new THREE.Mesh(geo, mat);
                mesh.rotation.x = Math.PI / 2;
                
                const isHigh = Math.random() > 0.8; 
                mesh.position.set(xPos, isHigh ? 2.5 : 0.6, coinZ);
                mesh.castShadow = true; scene.add(mesh);
                
                gameObjects.push({ mesh: mesh, type: 'coin', isObstacle: false, rotationSpeed: 0.1, lane: lane });
            }
        }

        function moveLeft() { if (currentLane > -1) { currentLane--; targetX = currentLane * -LANE_WIDTH; } }
        function moveRight() { if (currentLane < 1) { currentLane++; targetX = currentLane * -LANE_WIDTH; } }

        function jump() {
            const config = CHAR_CONFIG[selectedChar];
            // เช็คว่ากดกระโดดเกินโควต้าของแต่ละตัวหรือยัง
            if (jumpCount < config.maxJumps && !isDucking) { 
                isJumping = true; 
                jumpCount++; // นับว่าโดดไปกี่ครั้งแล้ว
                velocityY = config.jumpPower; 
                AudioSys.playSound('jump'); 
            }
        }

        function duck() {
            const config = CHAR_CONFIG[selectedChar];
            if (!config.canDuck) return; // หนูจี๊ดก้มไม่ได้!

            if (!isJumping && !isDucking) {
                isDucking = true; AudioSys.playSound('duck');
                // บีบตัวลงตามสัดส่วนของตัวละครนั้นๆ
                playerGroup.scale.set(config.scale, config.scale * 0.4, config.scale); 
                playerGroup.position.y = -0.3; 
                clearTimeout(duckTimeout);
                duckTimeout = setTimeout(() => { 
                    isDucking = false; 
                    playerGroup.scale.set(config.scale, config.scale, config.scale); 
                    playerGroup.position.y = 0; 
                }, 800);
            }
        }

        function setupEvents() {
            // คีย์บอร์ด
            document.addEventListener('keydown', (e) => {
                if (!isGameRunning) return;
                switch(e.code) {
                    case 'ArrowLeft': case 'KeyA': moveLeft(); break;
                    case 'ArrowRight': case 'KeyD': moveRight(); break;
                    case 'ArrowUp': case 'KeyW': case 'Space': jump(); break;
                    case 'ArrowDown': case 'KeyS': duck(); break;
                }
            });

            // ปัดหน้าจอ
            let touchStartX = 0, touchStartY = 0;
            document.addEventListener('touchstart', e => { touchStartX = e.changedTouches[0].screenX; touchStartY = e.changedTouches[0].screenY; });
            document.addEventListener('touchend', e => {
                if (!isGameRunning) return;
                let diffX = e.changedTouches[0].screenX - touchStartX, diffY = e.changedTouches[0].screenY - touchStartY;
                if (Math.abs(diffX) > Math.abs(diffY)) {
                    if (Math.abs(diffX) > 30) { if (diffX > 0) moveRight(); else moveLeft(); }
                } else {
                    if (Math.abs(diffY) > 30) { if (diffY < 0) jump(); else duck(); }
                }
            });

            window.addEventListener('resize', () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); });

            // UI Buttons
            document.getElementById('start-btn').addEventListener('click', startGame);
            document.getElementById('restart-btn').addEventListener('click', startGame);
            document.getElementById('home-btn').addEventListener('click', goHome);

            // ระบบเปิด-ปิด ห้อง Avatar
            document.getElementById('go-avatar-btn').addEventListener('click', () => {
                document.getElementById('start-screen').classList.add('hidden');
                document.getElementById('avatar-screen').classList.remove('hidden');
                isInAvatarRoom = true;
                pedestal.visible = true; // เปิดแท่นยืน
            });

            document.getElementById('back-main-btn').addEventListener('click', () => {
                document.getElementById('avatar-screen').classList.add('hidden');
                document.getElementById('start-screen').classList.remove('hidden');
                isInAvatarRoom = false;
                pedestal.visible = false; // ซ่อนแท่นยืน
            });

            // ระบบเลือกตัวละคร (ปรับปรุงให้คลิกตัวที่ปลดล็อคใหม่ได้ทันที)
            const avatarGrid = document.querySelector('.grid');
            avatarGrid.addEventListener('click', (e) => {
                const target = e.target.closest('button[data-char]');
                if (!target) return;
                
                // ถ้าเป็นช้าง แต่เหรียญยังไม่ครบ 250 ให้กดยังไงก็ไม่ติด!
                if (target.dataset.char === 'elephant' && totalCoins < 250) {
                    return; 
                }

                // ลบสถานะเลือกจากทุกปุ่ม
                document.querySelectorAll('button[data-char]').forEach(b => {
                    b.classList.remove('selected', 'border-orange-500', 'bg-orange-50');
                    // ใส่ขอบเทากลับให้ปุ่มปกติ (ยกเว้นปุ่มล็อค)
                    if(b.id !== 'elephant-btn' || totalCoins >= 250) b.classList.add('border-gray-200');
                });
                
                // แอดสถานะให้ปุ่มที่กด
                target.classList.add('selected', 'border-orange-500', 'bg-orange-50');
                target.classList.remove('border-gray-200');
                
                selectedChar = target.dataset.char;
                document.getElementById('char-desc').innerText = CHAR_CONFIG[selectedChar].name;
                
                // อัปเดตโมเดลในฉาก
                buildCharacter(selectedChar);
            });
        }

            // จัดการเปิด/ปิด HUD ของช้าง (โชว์จำนวนครั้งที่ชนได้)
            if (selectedChar === 'elephant') {
                document.getElementById('smash-hud').classList.remove('hidden');
                document.getElementById('smash-display').innerText = CHAR_CONFIG.elephant.maxSmash;
            } else {
                document.getElementById('smash-hud').classList.add('hidden');
            }

            isInAvatarRoom = false;
            pedestal.visible = false; // ซ่อนแท่นยืนตอนเล่นเกม

            // 👇 สิ่งที่พี่มิ้นเพิ่มให้! ล้างของเก่าทิ้งให้หมดเกลี้ยง จะได้ไม่เกิดบัคตายรัวๆ 👇
            gameObjects.forEach(obj => scene.remove(obj.mesh)); 
            gameObjects = [];
            envObjects.forEach(obj => scene.remove(obj)); 
            envObjects = [];

            // รีเซ็ตตัวนับเวลาเกิดของใหม่ จะได้ไม่โผล่มาชนหน้าตอนเริ่มเกมทันที
            spawnTimer = 0; 
            coinPatternTimer = 0; 
            treeSpawnTimer = 0;
            // 👆 จบส่วนที่แก้ไข 👆
            
            score = 0; distance = 0; gameSpeed = BASE_SPEED;
            currentLane = 0; targetX = 0;
            playerGroup.position.set(0, 0, 0); playerGroup.rotation.set(0,0,0);
            
            // รีเซ็ตสเกลตามตัวละครเผื่อค้างตอนตาย
            const cScale = CHAR_CONFIG[selectedChar].scale;
            playerGroup.scale.set(cScale, cScale, cScale);

            isJumping = false; isDucking = false; velocityY = 0; jumpCount = 0;
            currentSmashCount = 0; // รีเซ็ตจำนวนที่พุ่งชนเวลาเริ่มเกมใหม่
            
            document.getElementById('score-display').innerText = score;
            document.getElementById('distance-display').innerText = 0;
            
            isGameRunning = true; AudioSys.playBGM();
        }

        function goHome() {
            document.getElementById('game-over-screen').classList.add('hidden');
            document.getElementById('start-screen').classList.remove('hidden');
            updateHighScoreDisplay();
            updateStatsDisplay();
            
            // ทำให้ตัวละครตั้งขึ้น
            playerGroup.rotation.x = 0; 
            playerGroup.position.y = 0;
        }

        function gameOver(obstacleType) {
            isGameRunning = false; AudioSys.stopBGM(); AudioSys.playSound('crash');
            
            document.getElementById('hud').classList.add('hidden');
            document.getElementById('game-over-screen').classList.remove('hidden');
            
            let finalDist = Math.floor(distance);
            document.getElementById('final-coins').innerText = score;
            document.getElementById('final-dist').innerText = finalDist;
            
            // บันทึกเหรียญสะสมรวม 🪙
            totalCoins += score;
            localStorage.setItem('animalDashTotalCoins', totalCoins);
            updateUnlockProgress(); // อัปเดตหลอดความคืบหน้าของช้างทันที
            
            // บันทึกสถิติสูงสุด
            let isNewRecord = false;
            if (score > bestScore) { bestScore = score; localStorage.setItem('animalDashBestScore', bestScore); isNewRecord = true; }
            if (finalDist > bestDist) { bestDist = finalDist; localStorage.setItem('animalDashBestDist', bestDist); isNewRecord = true; }
            
            if(isNewRecord) document.getElementById('new-record-badge').classList.remove('hidden');
            else document.getElementById('new-record-badge').classList.add('hidden');

            if (obstacleType === 'rock') stats.rock++;
            else if (obstacleType === 'pit') stats.pit++;
            else if (obstacleType === 'vine') stats.vine++;

            localStorage.setItem('animalDashStatsRock', stats.rock);
            localStorage.setItem('animalDashStatsPit', stats.pit);
            localStorage.setItem('animalDashStatsVine', stats.vine);
            
            updateStatsDisplay();

            // ล้มลง
            playerGroup.rotation.x = -Math.PI / 2;
            playerGroup.position.y = 0.5; // ยกลอยนิดนึงไม่ให้จมดิน
        }

        function animate() {
            requestAnimationFrame(animate);
            if (isGameRunning) {
                gameSpeed += 0.0001; distance += (gameSpeed * 0.1);
                
                // อัปเดตแสงสี
                updateDayNightCycle();

                if (Math.floor(distance) % 5 === 0) document.getElementById('distance-display').innerText = Math.floor(distance);

                // Lerp การเคลื่อนที่ซ้ายขวา
                playerGroup.position.x += (targetX - playerGroup.position.x) * 0.15;
                
                // ระบบฟิสิกส์กระโดด
                if (isJumping) {
                    playerGroup.position.y += velocityY; 
                    
                    // คำนวณแรงโน้มถ่วงแยกตามประเภทตัวละคร (ไก่ร่อนช้า)
                    const config = CHAR_CONFIG[selectedChar];
                    const currentGravity = gravity * config.gravityScale;
                    velocityY += currentGravity; 
                    
                    // เพิ่มแอนิเมชันกระพือปีกเฉพาะตอนไก่กระโดด
                    if (playerGroup.userData.isChicken) {
                        const time = Date.now() * 0.025;
                        playerGroup.userData.wingL.rotation.z = Math.sin(time) * 0.5 - 0.2;
                        playerGroup.userData.wingR.rotation.z = -(Math.sin(time) * 0.5 - 0.2);
                    }

                    if (playerGroup.position.y <= 0) { 
                        playerGroup.position.y = 0; 
                        isJumping = false; 
                        jumpCount = 0; // ถึงพื้นแล้ว คืนสิทธิ์การกระโดด
                        velocityY = 0; 
                        
                        // เอาปีกไก่ลงเมื่อถึงพื้น
                        if (playerGroup.userData.isChicken) {
                            playerGroup.userData.wingL.rotation.z = 0;
                            playerGroup.userData.wingR.rotation.z = 0;
                        }
                    }
                }
                
                playerGroup.rotation.z = -(targetX - playerGroup.position.x) * 0.1;

                spawnTimer -= gameSpeed; coinPatternTimer -= gameSpeed; treeSpawnTimer -= gameSpeed;
                
                if (spawnTimer <= 0) { spawnObstacle(); spawnTimer = 20 + Math.random() * 20; }
                if (coinPatternTimer <= 0) { spawnCoins(); coinPatternTimer = 50 + Math.random() * 40; }
                if (treeSpawnTimer <= 0) { spawnTree(); treeSpawnTimer = 5 + Math.random() * 10; } 

                // การเช็คชน (Collision) อิงตาม BoundingBox 
                const playerBox = new THREE.Box3().setFromObject(playerGroup);
                
                // ปรับขอบเขต Hitbox ให้นุ่มนวลขึ้น
                const boxWidth = playerBox.max.x - playerBox.min.x;
                const boxDepth = playerBox.max.z - playerBox.min.z;
                playerBox.min.x += boxWidth * 0.2; 
                playerBox.max.x -= boxWidth * 0.2;
                playerBox.min.z += boxDepth * 0.2; 
                playerBox.max.z -= boxDepth * 0.2;
                // หดความสูงลงนิดหน่อยให้รอดเถาวัลย์ง่ายขึ้นเวลาขนาดหมิ่นเหม่
                playerBox.max.y -= 0.1; 

                // อัปเดตสิ่งของ
                for (let i = gameObjects.length - 1; i >= 0; i--) {
                    const obj = gameObjects[i];
                    obj.mesh.position.z -= gameSpeed;
                    
                    if (!obj.isObstacle) { 
                        obj.mesh.rotation.y += obj.rotationSpeed; 
                        obj.mesh.rotation.z += obj.rotationSpeed; 
                    }
                    
                    const objBox = new THREE.Box3().setFromObject(obj.mesh);
                    if (playerBox.intersectsBox(objBox)) {
                        if (obj.isObstacle) { 
                            // เช็คพลังความสามารถของช้าง (ชนหินและเถาวัลย์ได้)
                            if (selectedChar === 'elephant' && (obj.type === 'rock' || obj.type === 'vine') && currentSmashCount < CHAR_CONFIG.elephant.maxSmash) {
                                currentSmashCount++;
                                document.getElementById('smash-display').innerText = CHAR_CONFIG.elephant.maxSmash - currentSmashCount;
                                scene.remove(obj.mesh); 
                                gameObjects.splice(i, 1);
                                AudioSys.playSound('crash'); // เสียงชนของพัง
                                
                                // เพิ่มแอนิเมชันสั่นๆ ตอนชนนิดนึงเพื่ออรรถรส
                                playerGroup.rotation.z = 0.2 * (Math.random() > 0.5 ? 1 : -1);
                                setTimeout(() => { if(playerGroup) playerGroup.rotation.z = 0; }, 100);
                                
                                continue; // วิ่งต่อไปได้ชิลๆ!
                            } else {
                                gameOver(obj.type); 
                                break; 
                            }
                        } 
                        else {
                            score++; document.getElementById('score-display').innerText = score;
                            AudioSys.playSound('coin'); 
                            scene.remove(obj.mesh); 
                            gameObjects.splice(i, 1); 
                            continue;
                        }
                    }
                    if (obj.mesh.position.z < -10) { scene.remove(obj.mesh); gameObjects.splice(i, 1); }
                }

                // อัปเดตต้นไม้ประดับฉาก
                for (let i = envObjects.length - 1; i >= 0; i--) {
                    envObjects[i].position.z -= gameSpeed;
                    if (envObjects[i].position.z < -10) { scene.remove(envObjects[i]); envObjects.splice(i, 1); }
                }
            }
            
            // หน้าเริ่มเกม ให้ตัวละครหมุนโชว์ความเท่
            else if (!isGameRunning) {
                if (playerGroup) {
                    playerGroup.rotation.y += 0.015; // หมุนตัวตลอดเวลา
                    
                    if (isInAvatarRoom) {
                        // แก้บัคแอบหลังจอ! แกนซ้ายคือ X ที่เป็นบวก (+)
                        // หน้าจอคอม(>768px): ย้ายไปซ้าย (x=2.5), ดึงมาใกล้กล้องให้เห็นชัดๆ (z=-2)
                        // หน้าจอมือถือ: อยู่ตรงกลาง (x=0), ดึงมาใกล้กล้องให้เห็นชัดๆ (z=-3)
                        let targetAvatarX = window.innerWidth > 768 ? 2.5 : 0;
                        let targetAvatarZ = window.innerWidth > 768 ? -2 : -3; 
                        
                        playerGroup.position.x += (targetAvatarX - playerGroup.position.x) * 0.1;
                        playerGroup.position.z += (targetAvatarZ - playerGroup.position.z) * 0.1;
                        
                        // เลื่อนแท่นตามตัวละคร
                        pedestal.position.x = playerGroup.position.x;
                        pedestal.position.z = playerGroup.position.z;
                    } else {
                        // ถ้ากลับมาหน้า Start Screen ให้ตัวละครค่อยๆ เลื่อนกลับมาตรงกลาง
                        playerGroup.position.x += (0 - playerGroup.position.x) * 0.1;
                        playerGroup.position.z += (0 - playerGroup.position.z) * 0.1;
                    }
                }
            }

            renderer.render(scene, camera);
        }
